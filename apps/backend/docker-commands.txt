# stop / start / restart
docker stop cassandra
docker start cassandra
docker restart cassandra

# view logs
docker logs -f cassandra

# remove the container (keeps data in the named volume)
docker rm -f cassandra

# Set-up the 3-node cluster defined in docker-compose
docker-compose up -d

# Local dev (single-node)
export CASSANDRA_HOSTS=""   # or just leave unset
go run main.go

# Docker cluster (connects to all three nodes in your Docker network)
export CASSANDRA_HOSTS="cassandra1,cassandra2,cassandra3"
go run main.go

# Access Cassandra
docker exec -it cassandra1 cqlsh

# nuke data (careful!)
docker volume rm cassandra-data

# ----- RUN BUILD ------
docker system prune
cd server
GOOS=linux GOARCH=amd64 go build -o server ./main.go
cd ..
docker-compose up --build -d backend
docker-compose logs -f backend

# Stop running containers 
docker ps
docker stop $(docker ps -q)
